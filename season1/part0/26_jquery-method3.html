제이쿼리 메소드 - 요소 탐색

    $('선택자').함수( function(){ 
      $('선택요소').메소드();
    } );
  
    children() : 선택요소의 바로 아래 단계인 자식요소만 선택
    siblings() : 선택요소의 형제요소를 선택
    parent(),  find(), next(), prev()
