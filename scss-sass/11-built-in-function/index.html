scss에서 사용할 수 있는 내장 함수

this is h2

scss 공식 내장함수 종류 
    예시
    map-get() : // Map(변수) 내에서 특정 key의 value를 css로 반환(출력)
    ---------------------------------------
    $fonts: (
      kor: 'Noto Sans KR', 
      eng: 'Montserrat'
    );

    body {
      font-family: map-get($fonts, kor), sans-serif;
    }

    h2 {
      font-family: map-get($fonts, eng), sans-serif;
    }
    -------------------------------------------
    $color: (
      primary: #222,
      secondary: #eee,
      focus: tomato
    );

    body {
      background-color: map-get($color, secondary);
      color: map-get($color, primary);
    }
    h1 {
      color: map-get($color, focus);
    }
